cmake_minimum_required(VERSION 4.1)

project("scaner")

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED on)

add_executable(scaner main.cpp)

target_link_libraries(scaner PRIVATE lib)

add_custom_command(TARGET scaner POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    $<TARGET_FILE:lib>
    $<TARGET_FILE_DIR:scaner>
)
